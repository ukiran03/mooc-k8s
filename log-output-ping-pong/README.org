** 2.7. Stateful applications

#+begin_src bash
# begin, explicitly apply namespace (I think its a good practice)
kubectl apply -f namespace.yaml

# first, start postgres statefulset
kubectl apply -f pg-secret.yaml
kubectl apply -f pg-service.yaml
kubectl apply -f pg-statefulset.yaml

# finally, start the app
kubectl apply -f configmap.yaml
kubectl apply -f deployment.yaml
kubectl apply -f ingress.yaml
kubectl apply -f service.yaml
