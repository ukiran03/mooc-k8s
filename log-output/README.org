*** 1.1. Getting started

#+begin_src bash
# Build the local image
docker build -t log-output:v1.1 .

# Import the local image
k3d image import log-output:v1.1 -c k3s-default

# Create Deployment with local image
kubectl create deployment log-output-dep --image=log-output:v1.1

# Checking
kubectl get deployments
kubectl get pods

# Check Logs
kubectl logs -f log-output-dep-7c5d45c777-lpgmf
#+end_src
